How we start? (1)

Very simple!

[source, python]
----
import sys

from django.conf import settings
from django.conf.urls import url
from django.http import HttpResponse

settings.configure(
    DEBUG=True,
    SECRET_KEY="secret",
    ROOT_URLCONF=sys.modules[__name__],
)

def index(request):
    return HttpResponse("Hello World!")

urlpatterns = [
    url(r"^$", index, name="index"),
]
